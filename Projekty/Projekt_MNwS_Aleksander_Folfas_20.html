Za pomoc odpowiednich symulacji zbada odsetek odrzuce prawdziwej hipotezy gwnej (tzn. bd I rodzaju) w tecie istotnoci wspczynnika korelacji w przypadku, gdy dane pochodz z rozkadu o grubych ogonach (np. -Studenta o rnych stopniach swobody). W trakcie symulowania danych naley pamita, by dane speniay hipotez gwn. Porwna uzyskane wyniki dla wsp. korelacji Pearsona i Spearmana. Uzyskane wyniki naley opisa w zalenoci od:

  • liczby danych,
  • liczby stopni swobody generowanych rozkadw 2

def rejection_rate_correlation(n_samples, df, n_simulations=1000):
    pearson_rejections = 0
    spearman_rejections = 0

    for _ in range(n_simulations):
        # Generowanie danych z rozkadu t-Studenta o rnych stopniach swobody
        group1 = np.random.standard_t(df, size=n_samples)
        group2 = np.random.standard_t(df * 2, size=n_samples)

        # Test istotnoci wspczynnika korelacji Pearsona
        pearson_r, pearson_pvalue = stats.pearsonr(group1, group2)
        if pearson_pvalue < 0.05:
            pearson_rejections += 1

        # Test istotnoci wspczynnika korelacji Spearmana
        spearman_rho, spearman_pvalue = stats.spearmanr(group1, group2)
        if spearman_pvalue < 0.05:
            spearman_rejections += 1

    pearson_rejection_rate = pearson_rejections / n_simulations
    spearman_rejection_rate = spearman_rejections / n_simulations

    return pearson_rejection_rate, spearman_rejection_rate

# Parametry symulacji
n_samples_range = np.arange(10, 51, 10)
df_range = [1, 5, 10, 20, 30]
np.random.seed(411052)

pearson_rejection_rates = np.zeros((len(n_samples_range), len(df_range)))
spearman_rejection_rates = np.zeros((len(n_samples_range), len(df_range)))

# Symulacje dla rnych wartoci liczby danych i stopni swobody rozkadu t-Studenta
for i, n_samples in enumerate(n_samples_range):
    for j, df in enumerate(df_range):
        pearson_rejection_rate, spearman_rejection_rate = rejection_rate_correlation(n_samples, df)
        pearson_rejection_rates[i, j] = pearson_rejection_rate
        spearman_rejection_rates[i, j] = spearman_rejection_rate

# Wykresy dla testu istotnoci wspczynnika korelacji Pearsona
plt.figure(figsize=(10, 6))
for j, df in enumerate(df_range):
    plt.plot(n_samples_range, pearson_rejection_rates[:, j], '-o', label=f'df={df}')
plt.xlabel('Liczba prbek')
plt.ylabel('Odrzut testu Pearsona')
plt.title('Odrzut testu istotnoci wspczynnika korelacji Pearsona')
plt.legend()
plt.grid(True)
plt.show()
