Za pomoc odpowiednich symulacji zbada moc testu Komogorowa-Smirnowa w przypadku, gdy rozwaane prbki danych pochodz z rozkadw normalnych o tej samej wartoci oczekiwanej, ale o rnych odchyleniach standardowych. Rozway rne wartoci oczekiwane. Uzyskane wyniki naley przedstawi na odpowiednich wykresach ilustrujcych moce testw z uwzgldnieniem:

  • liczby danych (mona rozwaa rne liczebnoci w kadej z prbek),
  • odchyle standardowych.

# Lista wartoci odchylenia standardowego
std_values = [0.5, 1, 2, 4]

# Lista liczby danych (liczebnoci prbek)
n_values = np.arange(10, 1010, 10)

# Ziarno losowoci
np.random.seed(411052)

# Macierz do przechowywania wynikw
powers = np.zeros((len(std_values), len(n_values)))

# Przeprowadzenie symulacji
fig, axes = plt.subplots(nrows=2, ncols=2, figsize=(10, 10))
axes = axes.flatten()

for i, std in enumerate(std_values):
    for j, n in enumerate(n_values):
        # Wygenerowanie prbek danych z rozkadw normalnych
        data = np.random.normal(0, std, n)
        
        # Obliczenie mocy testu Komogorowa-Smirnowa
        _, p_value = kstest(data, 'norm')
        power = 1 - p_value
        
        # Zapisanie wyniku
        powers[i, j] = power
    
    # Wykres mocy testu w zalenoci od liczby danych i odchylenia standardowego
    ax = axes[i]
    ax.ymin = 0
    ax.ymax = 1
    ax.plot(n_values, powers[i], label=f"Std={std}", color=f"C{i}")
    ax.set_xlabel('Liczba danych')
    ax.set_ylabel('Moc testu')
    ax.set_title(f'Moc testu Komogorowa-Smirnowa (std={std})')
    ax.legend()
    ax.grid(True)

plt.tight_layout()
plt.show()
